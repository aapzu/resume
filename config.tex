\newcommand\languageknowledge[1]
  {%
    \hbox
      {%
        \languagecount=0
        \loop\ifnum\languagecount<#1
          \advance\languagecount1
          \textcolor{sectionrectanglecolor}{$\bullet$}%
        \repeat
        \loop\ifnum\languagecount<5
          \advance\languagecount1
          \textcolor{nolanguagecolor}{$\bullet$}%
        \repeat
      }%
  }
